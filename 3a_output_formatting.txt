import requests
import json

def emotion_detector(text_to_analyze):
    # The URL provided by your assignment instructions
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    
    # The required headers indicating which model to use
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    
    # The JSON payload structure containing the text variable
    myobj = { "raw_document": { "text": text_to_analyze } }
    
    # Sending a POST request to the Watson NLP API
    response = requests.post(url, json=myobj, headers=headers)
    
    # Convert the response text into a dictionary using the json library functions
    formatted_response = json.loads(response.text)
    
    # Extract the required set of emotions along with their scores
    emotions = formatted_response['emotionPredictions'][0]['emotion']
    anger_score = emotions['anger']
    disgust_score = emotions['disgust']
    fear_score = emotions['fear']
    joy_score = emotions['joy']
    sadness_score = emotions['sadness']
    
    # Write the code logic to find the dominant emotion
    emotion_list = {
        'anger': anger_score,
        'disgust': disgust_score,
        'fear': fear_score,
        'joy': joy_score,
        'sadness': sadness_score
    }
    dominant_emotion = max(emotion_list, key=emotion_list.get)
    
    # Format output according to the assignment instructions
    result = {
        'anger': anger_score,
        'disgust': disgust_score,
        'fear': fear_score,
        'joy': joy_score,
        'sadness': sadness_score,
        'dominant_emotion': dominant_emotion
    }
    
    return result
